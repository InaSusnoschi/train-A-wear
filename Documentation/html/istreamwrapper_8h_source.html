92  void Read() {
93  if (current_ < bufferLast_)
94  ++current_;
95  else if (!eof_) {
96  count_ += readCount_;
97  readCount_ = bufferSize_;
98  bufferLast_ = buffer_ + readCount_ - 1;
99  current_ = buffer_;

102  readCount_ = static_cast<size_t>(stream_.gcount());
